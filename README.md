# Code2

# pobranie
git clone https://github.com/LukiRage/Code2

# instalacja
cd Code2 <br>
tar -zxvf code2-2.0.tar.gz -C (lokalizacja gdzie ma zostać rozpakowany plik) <br>
cd (lokalizacja wypakowanego pliku)/code2-2.0 <br>
./configure <br>
sudo make <br>
sudo make install <br>
